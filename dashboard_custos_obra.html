Configuração — Google Sheets via Apps Script
Configure uma vez para salvar o histórico automaticamente.
1Abra script.google.com → "Novo projeto"
2Cole o código abaixo (clique para copiar) e salve com Ctrl+S
3Implantar → Nova implantação → App da Web → Executar como: Eu → Acesso: Qualquer pessoa
4Copie a URL do App da Web e cole abaixo
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.openById(data.sheetId); var sheet = ss.getSheetByName('Historico') || ss.insertSheet('Historico'); if (sheet.getLastRow() === 0) { sheet.appendRow(['Data/Hora','Obra','Categoria','Item','Orçado (R$)','Realizado (R$)','Desvio (R$)','Venda (R$)','Imposto (R$)','Lucro Liq (R$)','Margem (%)']); sheet.getRange(1,1,1,11).setFontWeight('bold').setBackground('#e8f0fe'); } data.rows.forEach(function(r){ sheet.appendRow(r); }); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,erro:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { var sheetId = e.parameter.sheetId; var ss = SpreadsheetApp.openById(sheetId); var sheet = ss.getSheetByName('Historico'); if (!sheet) return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify({rows:sheet.getDataRange().getValues()})).setMimeType(ContentService.MimeType.JSON); }
